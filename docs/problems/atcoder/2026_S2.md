## AtCoder Beginner Contest [#446](https://atcoder.jp/contests/abc446)

E - Multiple-Free Sequences

由題目可以知道:

\[
\begin{array}{ll}
    S_1 = y, S_2 = x \rightarrow (S_2, S_1) = (x, y) \\
    (S_2, S_1) \rightarrow (S_3, S_2) = (Ax + By, x)
\end{array}
\]

由上面可以得到矩陣轉移方程:

$$
\begin{bmatrix}
A && B\\
1 && 0
\end{bmatrix}
\begin{bmatrix}
S_2 \\
S_1
\end{bmatrix}
=\begin{bmatrix}
AS_1 + BS_2\\
S_1
\end{bmatrix}
=\begin{bmatrix}
Ax + By\\
x
\end{bmatrix}
=\begin{bmatrix}
S_3\\
S_2
\end{bmatrix}
$$

$$
\begin{bmatrix}
A && B\\
1 && 0
\end{bmatrix}
\begin{bmatrix}
S_{i+1} \\
S_i
\end{bmatrix}
=\begin{bmatrix}
AS_{i+1} + BS_{i}\\
S_{i+1}
\end{bmatrix}
=\begin{bmatrix}
S_{i+2}\\
S_{i+1}
\end{bmatrix}
$$

簡單來說每次的轉移會是:

$$(x, y) \rightarrow (Ax + By, x)$$

題目所求 $S_i$ 不能為 M 的倍數因此，可以透過 $\pmod{M}$ 將問題轉化為，求 $S_i \pmod{M} \neq 0$ 的點對個數。

$$(x, y) \rightarrow (Ax + By \pmod{M}, x)$$

由題目的限制可以知道 $0\le x,y < M \le 10^3$。最多會有 $m^2$ 種點對，且當 $(Ax + By \pmod{M}, x)$ 時可以知道一定會是 $m^2$ 種的其中一種。

對於每次 $(x, y) \rightarrow (Ax + By \pmod{M}, x)$ 可以建一個反邊。最後由 $(u, 0), (0, v), (0, 0)$ 做 dfs 把所有遍歷到的節點就是題目所求。


F - Reachable Set 2

VP的時候寫了一個DFS結果TLE，然後改BFS就過了，atcoder 寫dfs的時候要注意!

盡量用 global(少用lambda?)，以及使用陣列，和 vis 去重，不要使用set。

```cpp
int nbr[MX], vis[MX];
vector<int> g[MX];
int cnt_nbr = 0, cnt_vis = 0;
void dfs(int x, int limit) {
    vis[x] = 1;
    cnt_vis++;
    for(auto &y: g[x]) {
        if(y < limit && !vis[y]) {
            dfs(y, limit);
        }
        if(!nbr[y]) {
            nbr[y] = true;
            cnt_nbr++;
        }
    }
};

void solve() {
    int n, m, u, v;
    cin >> n >> m;
    for(int i = 0; i < m; ++i) {
        cin >> u >> v;
        if(u == v) continue;
        g[u].push_back(v);
    }    

    for(int i = 1; i <= n; ++i) {
        if(i == 1) {
            nbr[i] = 1;
            cnt_nbr++;
        }
        if(nbr[i]) dfs(i, i);
        if(cnt_vis == i) cout<<cnt_nbr - cnt_vis<<endl;
        else cout<<-1<<endl;

    }
}
```

D - No-Subsequence Substring

這題要計算所有在 $S$ 中不包含構成 $T$ 子序列 t 的子字串個數。

可以透過計算全部方案減去包含 $T$ 子序列的子字串 $s_T$ 的數量: $all - cnt(s_T)$

當固定右端點時，可以觀察到越往左越有機會能夠找到包含 T 的子字串 $s_T$，而一旦找到 $s_T$ 則 $s_T$ 左邊的端點也可以與當前右端點構成 $s_T$。

可以向右遍歷陣列，不斷計算更新可以構成 $s_T$ 的最近位置。

```cpp
void solve() {
    string s, t;
    ll ans = 0;
    cin >> s >> t;
    int n = t.size();
    ll m = s.size();
    vector<int> cnt(n);
    for(int j = 0; j < m; j++) {
        auto &c = s[j];
        int ch = c - 'a';
        for(int i = n - 1; i >= 0; --i) {
            if(t[i] == c) {
                if(i == 0) {
                    cnt[i] = j + 1;
                    continue;
                }
                cnt[i] = cnt[i - 1];
            }
        }
        ans += cnt[n - 1];
    }
    cout<<m * (m + 1) / 2 - ans<<endl;
}
```

F - Exactly K Steps 2

由 Floyd 算法出發，可以知道:
```
for k = 1 to n
    for i = 1 to n
        for j = 1 to n
            M'[i][j] = min(M'[i][j], M[i][k] + M[k][j])
```

由 i 經由 k 至 j，產生出來的矩陣為 $D'$，$D[i][k]$ 為從 i 至 k 為走一步的花費，則 $D'[i][j]$ 為從 i 至 j 走兩步的花費。

由矩陣乘法可以得到

```
for k = 1 to n
    for i = 1 to n
        for j = 1 to n
            D'(i, j) += D(i, k) * D(k, j)
```

可以發現 Floyd 跟矩陣乘法很相似，但是要使用矩陣乘法要滿足 "結合律" 和 "交換律"，假設 $D'(i, j) = min(D(i, k) + D(k, j))$ 的 $min$ 操作表示為 $\oplus$

1. 結合律
   $(D_{a} \oplus D) \oplus D_{b} = D_{a} \oplus (D \oplus D_{b})$

2. 交換律
    $D_{a} \oplus D_{b} = D_{b} \oplus D_{a}$


主要觀察 $min$ 操作是否能夠符合結合律和交換律，若是可以就能使用矩陣快速冪計算。

```cpp
const ll infll = LLONG_MAX / 2;
using Matrix_t = vector<vector<long long>>;
Matrix_t mul(Matrix_t &a, Matrix_t &b) {
    const int m = a.size(), n = b[0].size();
    Matrix_t mat = Matrix_t(m, vector<long long>(n, infll));
    for(int k = 0; k < n; ++k) {
        for(int i = 0; i < n; ++i) {
            for(int j = 0; j < n; ++j) {
                mat[i][j] = min(mat[i][j], a[i][k] + b[k][j]);
            }
        }
    }
    return mat;
}

Matrix_t mat_qpow(Matrix_t &x, int n) {
    Matrix_t ans = x;
    n--;
    while(n) {
        if(n & 1) ans = mul(x, ans);
        x = mul(x, x);
        n >>= 1;
    }
    return ans;
}

void solve() {
    int n, k;
    cin >> n >> k;
    Matrix_t cost(n, vector<ll>(n));
    for(int i = 0; i < n; ++i) {
        for(int j = 0; j < n; ++j) {
            cin >> cost[i][j];
        }
    }
    cost = mat_qpow(cost, k);
    for(int s = 0; s < n; ++s) cout<<cost[s][s]<<endl;
}
```

E - A += v

這題可以用畫圖的方式幫助理解，假設 $A_i = 1, 3, 3, 3, 3, 5, 5, 5$，把 $cnt(A_i)$ 按照 x-y 座標作圖可以得到長方圖:

```
**: 表示在集合中出現一次
--: 表示需要操作後才會出現
cnt(A)
|  -- -- ** -- --
|  -- -- ** -- **
|  ** -- ** -- **
|  ** -- ** -- **
--|--|--|--|--|--|---->A
   1  2   3  4  5 
```

可以觀察到 2, 4 需要先進行操作加入序列當中才可以使用 1 進行操作，而 2, 4 的操作順序會按照大小輪流，可以得到一個結論:

可以將 $cnt(A_i)$ 進行分組，可以按照分組離線回答詢問。

```cpp
template<typename T>
class FenwickTree {
    .......
    //求第 k 大
    //從 [1th, ... nth]
    T kth(int k) {
        int l = 0, r = tree.size();
        while(l + 1 < r) {
            int m = (l + r) / 2;
            T sum = pre(m);
            (sum >= k ? r : l) = m;
        }
        return r;
    }
};

struct Query_t{
    ll val;
    int id;
};

void solve() {
    int n, m, q;
    cin >> n >> m;
    FenwickTree<int> t(mapping(m));
    vector<int> fq(m + 1), a(n + 1);
    map<int, vector<int>> cnt;
    cnt[0] = {};
    for(int i = 1; i <= n; ++i) {
        cin >> a[i];
        fq[a[i]]++;
    }
    cin >> q;
    vector<Query_t> queries(q);
    
    for(int i = 0; i < q; ++i) {
        auto &[val, id] = queries[i];
        cin >> val;
        id = i;
    }
    ranges::sort(queries, [](const auto &a, const auto &b) {
        return a.val < b. val;
    });
    for(int x = 1; x <= m; ++x) {
        cnt[fq[x]].pb(x);
    }
    int i = 0;
    vector<int> ans(q);
    for(;i < q && queries[i].val <= n; i++) {
        auto &[val, id] = queries[i];
        ans[id] = a[val];
    }
    vector<ll> phases;
    int sum = n, suf = 0;
    
    for(auto it = cnt.rbegin(); next(it) != cnt.rend(); ++it) {
        auto [H, st] = *it;
        auto [nxtH, _] = *next(it);
        ll rs = n + 1LL * H * m - sum;
        phases.pb(rs);
        suf += st.size();
        sum -= 1LL * (H - nxtH) * suf;
    }
    if(n < phases.back()) phases.pb(n);
    ranges::reverse(phases);
    
    int j = 0;
    for(auto it = cnt.begin(); it != cnt.end() && j  + 1 < phases.size(); it++, j++) {
        auto &[H, st] = *it;
        for(auto &x: st) {
            t.update(x, 1);
        }
        int sz = t.pre(mapping(m));
        for(;i < q && queries[i].val <= phases[j + 1]; ++i) {
            auto &[val, id] = queries[i];
            int k = t.kth(((val - phases[j] - 1 + sz) % sz) + 1);
            ans[id] = k;
        }
    }
    for(;i < q; ++i) {
        auto &[val, id] = queries[i];
        int k = (val - phases[j] - 1 + m) % m + 1;
        ans[id] = k;
    }
    for(int j = 0; j < q; ++j) {
        cout<<ans[j]<<endl;
    }
}
```

F - Grid Clipping

這題想到用全部方塊減去黑色方塊所造成的貢獻，可以得到最後的答案。但是對於一堆點要算他們產生的面積，這部分就不知道如何計算。結果最後是需要動態開點線段樹+永久化標記。

實作的小細節為題目是網格座標，要套模板的話要把網格座標轉換成 x-y 軸座標。

[P5490 【模板】扫描线 & 矩形面积并](https://www.luogu.com.cn/problem/P5490)

```cpp
#define lc(node) tree[node].l
#define rc(node) tree[node].r

const int SZ = 8'000'000;
template<typename T>
class DynamicLazySegTree {
    struct Node {
        int l, r;
        T val;
        T tag;
    };
    inline static Node tree[SZ];
    const T INIT_TAG = 0;
    int empty;
    int root;
    int cnt;
    int rangeL;
    int rangeR;
    int newNode() {
        if(cnt >= SZ) return empty;
        int node = cnt++;
        tree[node].l = tree[node].r = empty;
        tree[node].val = 0;
        tree[node].tag = INIT_TAG;
        return node;
    }
    T merge_val(T &a, T &b) {
        return (a + b);
    }
    void ensure(int &node) {
        if (node == empty) {
            node = newNode();
        }
    }
    void maintain(int node, T l, T r) {
        if(tree[node].tag)
            tree[node].val = r - l + 1;
        else
            tree[node].val = merge_val(tree[lc(node)].val, tree[rc(node)].val);
    }
    void update(int &node, int l, int r, int ql, int qr, T val) {
        ensure(node);
        if (ql <= l && r <= qr) {
            tree[node].tag += val;
            maintain(node, ql, qr);
            return;
        }
        int m = l + (r - l) / 2;
        if(qr <= m) update(lc(node), l, m, ql, qr, val);
        else if(ql > m) update(rc(node), m + 1, r, ql, qr, val);
        else {
            update(lc(node), l, m, ql, m, val);
            update(rc(node), m + 1, r, m + 1, qr, val);
        }
        maintain(node, l, r);
    }
    T query(int node, int l, int r, int ql, int qr) {
        if (node == empty) return 0;
        if(ql <= l && r <= qr) {
            return tree[node].val;
        }
        int m = l + (r - l) / 2;
        if(qr <= m) return query(lc(node), l, m, ql, qr);
        else if(ql > m) return query(rc(node), m + 1, r, ql, qr);
        T rsL = query(lc(node), l, m, ql, m);
        T rsR = query(rc(node), m + 1, r, m + 1, qr);
        return merge_val(rsL, rsR);
    }

public:
    DynamicLazySegTree(int l, int r)
        : cnt(0), rangeL(l), rangeR(r) {
        empty = cnt++;
        tree[empty].l = empty;
        tree[empty].r = empty;
        tree[empty].val = 0;
        tree[empty].tag = INIT_TAG;
        root = empty;
    }
    void update(int ql, int qr, T val) {
        if (ql > qr) return;
        if (ql < rangeL || qr > rangeR) return;
        update(root, rangeL, rangeR, ql, qr, val);
    }
    T query(int ql, int qr) {
        if (ql > qr) return 0;
        if (ql < rangeL || qr > rangeR) return 0;
        return query(root, rangeL, rangeR, ql, qr);
    }
};
void solve() {
    int n, H, W, h, w, r, c;
    cin >> H >> W >> h >> w >> n;
    H = (H - h + 1);
    W = (W - w + 1);
    ll ans = 1LL * H * W;
    DynamicLazySegTree<ll> t(1, H + 1);
    map<int, vector<vector<int>>> pos;
    for(int i = 0; i < n; ++i) {
        cin >> r >> c;
        r++, c++;
        int rr = max(1, r - h), cc = max(1, c - w);
        chmin(r, H + 1), chmin(c, W + 1);
        pos[cc].pb({rr, r - 1, 1});
        pos[c].pb({rr, r - 1, -1});
    }
    int pre = pos.begin()->first;
    for(auto &[x, v]: pos) {
        int width = x - pre;
        int height = t.query(1, H + 1);
        ans -= 1LL * width * height;
        for(auto &p: v) {
            int l = p[0], r = p[1], opt = p[2];
            t.update(l, r, opt);
        }
        pre = x;
    }
    cout<<ans<<endl;
}
```

E - Sparse Range

一開始很自然地想到固定右端點，只要找最長的合法左端點即可，此時左右端點內構成的區間皆為合法。但是中間思考的時候方向做錯，想成固定又端點後要找最近 $|x_r - x_l| < D$ 的左端點，因此使用線段樹結果一直 WA，最後透過這個例子發現錯誤

```
4 1
3 3 3 1
```
重新思考後發現要找的是一個區間，區間內的相鄰元素距離至少要 D ，因此問題變成滑動窗口的問題，窗口為維護一個有序集合，集合中相鄰元素的差距 $\ge D$ ，一但有元素 $ < D$ 那就移動左端點直到窗口合法。

官解是用固定左端點的寫法。

void solve() {
    int n, d, xm, mx = 0;
    cin >> n >> d;
    vector<int> a(n);
    readv(n, a[i]);
    ll ans = 0;
    multiset<int> st;
    st.insert(inf * 2);
    st.insert(-inf);
    auto check = [&](int x) -> int {
        auto it = st.lower_bound(x);
        if(*next(it) - *it < d) return true;
        if(*it - *prev(it) < d) return true;
        return false;
    };
    for(int l = 0, r = 0; r < n; ++r) {
        int x = a[r];
        st.insert(x);
        while(l < r && check(x)) {
            st.erase(st.find(a[l++]));
        }
        ans += r - l + 1;
    }
    cout<<ans<<endl;
}
